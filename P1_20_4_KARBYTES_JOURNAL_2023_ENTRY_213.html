Write a C++ program which creates an int type array of 100 elements, stores a random integer no larger than 99 in each element of that array, and which prints the array values to the command line.

#include < iostream >
#include < cstdlib >
#include < ctime >

int main()
{
    // Create an int array of 100 elements
    int arr[100];

    // Seed random number generator
    srand(time(0));

    // Generate a random number no larger than 99 and store in each element of the array
    for (int i = 0; i < 100; i++)
    {
        arr[i] = rand() % 100; 
    }

    // Print the array values
    for (int i = 0; i < 100; i++)
    {
        std::cout << arr[i] << " ";
    }
    std::cout << std::endl;
    
    return 0;
}
